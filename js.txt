    const parseAmount = (val) => { 
        if (!val) return 0; 
        const cleanVal = String(val).replace(/[^0-9.-]+/g, ''); 
        const num = parseFloat(cleanVal); 
        return isNaN(num) ? 0 : num; 
    };

    const getHourFromTime = (timeStr) => {
        if (!timeStr) return null;
        const match = String(timeStr).trim().match(/^(\d{1,2}):/);
        if (match) return parseInt(match[1], 10);
        return null;
    };

    const formatMoney = (amount) => { 
        return amount.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 }); 
    };

    const formatDateTime = () => {
        const now = new Date();
        const d = now.getDate().toString().padStart(2, '0');
        const m = (now.getMonth() + 1).toString().padStart(2, '0');
        const y = now.getFullYear() + 543; // Buddhist Year
        const h = now.getHours().toString().padStart(2, '0');
        const min = now.getMinutes().toString().padStart(2, '0');
        const s = now.getSeconds().toString().padStart(2, '0');
        return `${d}/${m}/${y} ${h}:${min}:${s}`;
    };

    const parseDateLocal = (dateStr) => {
        if (!dateStr) return null;
        let dStr = String(dateStr).split(',')[0].split(' ')[0].split('T')[0].trim();
        if (dStr.includes('/')) {
            const p = dStr.split('/');
            if (p.length === 3) {
                if (p[2].length >= 4) return new Date(parseInt(p[2]), parseInt(p[1]) - 1, parseInt(p[0]));
                if (p[0].length >= 4) return new Date(parseInt(p[0]), parseInt(p[1]) - 1, parseInt(p[2]));
            }
        } else if (dStr.includes('-')) {
            const p = dStr.split('-');
            if (p.length === 3) {
                if (p[0].length >= 4) return new Date(parseInt(p[0]), parseInt(p[1]) - 1, parseInt(p[2]));
                if (p[2].length >= 4) return new Date(parseInt(p[2]), parseInt(p[1]) - 1, parseInt(p[0]));
            }
        }
        const d = new Date(dStr);
        if (!isNaN(d.getTime())) {
            d.setHours(0, 0, 0, 0);
            return d;
        }
        return null;
    };

    const formatTime = (timeStr) => {
        if (!timeStr) return "";
        let t = String(timeStr).trim().split(' ')[0];
        const parts = t.split(':');
        if (parts.length >= 2) {
            return `${parts[0].padStart(2, '0')}:${parts[1].padStart(2, '0')}`;
        }
        return "";
    };

                    <path d={`M ${strokeWidth / 2} ${cy} A ${radius} ${radius} 0 0 1 ${width - strokeWidth / 2} ${cy}`} fill="none" stroke="#f1f5f9" strokeWidth={strokeWidth} strokeLinecap="round" />

    const LeafletMap = ({ data, mapFocus }) => {
        const mapContainerRef = useRef(null);
        const mapInstanceRef = useRef(null);
        const markersLayerRef = useRef(null);
        const markerRefs = useRef({});

        useEffect(() => {
            if (!mapContainerRef.current || mapInstanceRef.current || !window.L) return;
            try {
                const map = L.map(mapContainerRef.current, { zoomControl: false }).setView([8.5, 99.0], 7);
                L.control.zoom({ position: 'topleft' }).addTo(map);
                L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '&copy; OpenStreetMap' }).addTo(map);
                markersLayerRef.current = L.layerGroup().addTo(map);
                mapInstanceRef.current = map;
                setTimeout(() => { map.invalidateSize(); }, 200);
            } catch (e) {}
            return () => { if (mapInstanceRef.current) { mapInstanceRef.current.remove(); mapInstanceRef.current = null; } };
        }, []);

        useEffect(() => {
            if (!mapInstanceRef.current || !markersLayerRef.current) return;
            markersLayerRef.current.clearLayers();
            markerRefs.current = {};

            const aggregatedPoints = {};
            data.forEach(point => {
                const latVal = String(point.atm_lat || '').replace(/[^0-9.-]/g, '');
                const lonVal = String(point.atm_lon || '').replace(/[^0-9.-]/g, '');
                const lat = parseFloat(latVal);
                const lon = parseFloat(lonVal);

                if (!isNaN(lat) && !isNaN(lon) && lat !== 0 && lon !== 0) {
                    const key = `${lat.toFixed(4)},${lon.toFixed(4)}`;
                    if (!aggregatedPoints[key]) {
                        let extraInfo = '';
                        if (point.source_type === 'ATM' || point._source_type === 'ATM') {
                            extraInfo = point.atm_name || point['ชื่อเครื่อง ATM'] || '';
                        } else {
                            extraInfo = point.branch_address || point['ที่อยู่สาขา'] || '';
                        }

                        aggregatedPoints[key] = {
                            lat, lon, count: 0, amount: 0,
                            location: point.atm_location || 'ไม่ระบุสถานที่',
                            province: point.atm_province || 'ไม่ระบุ',
                            extraInfo: extraInfo,
                            sourceType: point.source_type || point._source_type,
                            bankCodes: new Set()
                        };
                    }
                    aggregatedPoints[key].count += 1;
                    aggregatedPoints[key].amount += parseAmount(point.transfer_amount);
                    if (point.atm_bank_code && point.atm_bank_code !== 'ไม่ระบุ') {
                        aggregatedPoints[key].bankCodes.add(point.atm_bank_code);
                    }
                }
            });

            Object.values(aggregatedPoints).forEach(group => {
                let color = '#3b82f6'; let radius = 10;
                if (group.count >= 10) { color = '#ef4444'; radius = 24; }
                else if (group.count >= 5) { color = '#f97316'; radius = 16; }

                const marker = L.circleMarker([group.lat, group.lon], {
                    radius, fillColor: color, color: color, weight: 4, opacity: 0.3, fillOpacity: 0.6
                });
                const center = L.circleMarker([group.lat, group.lon], {
                    radius: radius / 2, fillColor: color, color: '#fff', weight: 1, opacity: 1, fillOpacity: 1
                });

                const extraInfoLabel = group.sourceType === 'ATM' ? 'ชื่อเครื่อง ATM' : 'ที่อยู่สาขา';

                        return `<span style="background-color: ${bgColor}; color: white; padding: 2px 6px; border-radius: 4px; font-size: 10px; font-weight: bold; margin-left: 4px; display: inline-block; margin-bottom: 2px;">${bank}</span>`;
                    }).join('');

                    bankHtml = `
                        <tr style="border-bottom: 1px dashed #e2e8f0;">
                            <td style="padding: 6px 0; color: #64748b; vertical-align: middle;">ธนาคาร</td>
                            <td style="padding: 6px 0; text-align: right;">${badges}</td>
                        </tr>
                    `;
                }

                const popupTableHtml = `
                    <div style="font-family: 'Sarabun', sans-serif; min-width: 250px;">
                        <div style="background-color: #1e293b; color: white; padding: 10px 12px; font-weight: bold; font-size: 13px; line-height: 1.3;">
                            <div style="display:flex; align-items:center; gap:6px;">
                                <svg xmlns="http://www.w3.org/2000/svg" width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M20 10c0 6-8 12-8 12s-8-6-8-12a8 8 0 0 1 16 0Z"/><circle cx="12" cy="10" r="3"/></svg>
                                ${group.location}
                            </div>
                        </div>
                        <div style="padding: 12px; background-color: #ffffff;">
                            <table style="width: 100%; font-size: 12px; color: #334155; border-collapse: collapse;">
                                ${bankHtml}
                                ${extraInfoHtml}
                                <tr style="border-bottom: 1px dashed #e2e8f0;">
                                    <td style="padding: 6px 0; color: #64748b;">จังหวัด</td>
                                    <td style="padding: 6px 0; text-align: right; font-weight: bold;">${group.province}</td>
                                </tr>
                                <tr style="border-bottom: 1px dashed #e2e8f0;">
                                    <td style="padding: 6px 0; color: #64748b;">ความถี่ (ครั้ง)</td>
                                    <td style="padding: 6px 0; text-align: right; font-weight: bold; color: #0f172a;">${group.count}</td>
                                </tr>
                                <tr>
                                    <td style="padding: 6px 0; color: #64748b;">ยอดรวม (บาท)</td>
                                    <td style="padding: 6px 0; text-align: right; font-weight: bold; color: #e11d48; font-size: 13px;">${group.amount.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</td>
                                </tr>
                            </table>
                        </div>
                    </div>
                `;

                const hoverTooltipHtml = `
                    <div style="font-family: 'Sarabun', sans-serif; text-align: center; min-width: 120px;">
                        <div style="font-weight: bold; color: #1e293b; font-size: 12px; border-bottom: 1px solid #e2e8f0; padding-bottom: 4px; margin-bottom: 4px;">${group.location}</div>
                        <div style="font-size: 11px; color: #64748b;">
                            ความถี่: <span style="font-weight: bold; color: #4f46e5;">${group.count} ครั้ง</span>
                        </div>
                        <div style="font-size: 11px; color: #64748b;">
                            ยอดรวม: <span style="font-weight: bold; color: #e11d48;">${group.amount.toLocaleString('en-US')} ฿</span>
                        </div>
                    </div>
                `;

                marker.bindPopup(popupTableHtml, { className: 'custom-popup', offset: [0, -radius] });
                marker.bindTooltip(hoverTooltipHtml, { direction: 'top', className: 'custom-map-tooltip', offset: [0, -radius] });

                markersLayerRef.current.addLayer(marker);
                markersLayerRef.current.addLayer(center);

                markerRefs.current[`${group.lat.toFixed(4)},${group.lon.toFixed(4)}`] = marker;
            });
        }, [data]);

        useEffect(() => {
            if (mapFocus && mapFocus.lat && mapFocus.lon && mapInstanceRef.current) {
                mapInstanceRef.current.flyTo([mapFocus.lat, mapFocus.lon], 13, { animate: true, duration: 1.5 });
                const key = `${parseFloat(mapFocus.lat).toFixed(4)},${parseFloat(mapFocus.lon).toFixed(4)}`;
                const targetMarker = markerRefs.current[key];
                if (targetMarker) {
                    setTimeout(() => { targetMarker.openPopup(); }, 500);
                }
            }
        }, [mapFocus]);

        return <div ref={mapContainerRef} className="w-full h-full z-0 rounded-lg shadow-inner min-h-[400px]" />;
    };

                {/* 🚨 ใหม่ล่าสุด! Custom Modal สำหรับแจ้งเตือน Error สวยๆ */}
                {customAlert.isOpen && (
                    <div className="fixed inset-0 bg-slate-900/60 z-[500] flex items-center justify-center p-4 backdrop-blur-sm transition-all animate-in fade-in">
                        <div className="bg-white rounded-2xl shadow-2xl w-full max-w-sm flex flex-col overflow-hidden animate-in zoom-in-95 duration-200">
                            <div className="p-6 text-center border-b border-slate-100 relative">
                                <div className="w-16 h-16 bg-red-50 text-red-500 rounded-full flex items-center justify-center mx-auto mb-4 shadow-inner">
                                    <Icons.AlertTriangle size={32} />
                                </div>
                                <h3 className="text-xl font-bold text-slate-800">{customAlert.title}</h3>
                                <p className="text-sm text-slate-600 mt-3 leading-relaxed">{customAlert.message}</p>
                            </div>
                            <div className="p-5 bg-slate-50 flex justify-center">
                                <button 
                                    onClick={() => setCustomAlert({ ...customAlert, isOpen: false })} 
                                    className="w-full py-3 rounded-xl font-bold text-white bg-indigo-600 hover:bg-indigo-700 transition-colors shadow-md"
                                >
                                    ตกลง / เข้าใจแล้ว
                                </button>
                            </div>
                        </div>
                    </div>
                )}

            </div>
        );
    };

    const root = ReactDOM.createRoot(document.getElementById('root'));
    root.render(<Dashboard />);
};

if (document.readyState === 'loading') { document.addEventListener('DOMContentLoaded', runDashboard); } else { runDashboard(); }
